99 void Tree::generateACube(const glm::vec3 &localPosition, float texture) {
100  for (int v = 0; v < cube.getVertexCount(); ++v) {
101  ShapeVertex shapeVertex;
102  const ShapeVertex &currentCubeVertex = cube.getDataPointer()[v];
103  double vertexX = currentCubeVertex.position.x / 2.;
104  // The cube is defined from -1 to 1 so it's size 2. Divide by two to get a
105  // 1x1x1 size cube
106  double vertexY = currentCubeVertex.position.y / 2.;
107  double vertexZ = currentCubeVertex.position.z / 2.;
108 
109  shapeVertex.position =
110  vec3(vertexX + positionTranslation.x + localPosition.x,
111  vertexY + positionTranslation.y + localPosition.y,
112  vertexZ + positionTranslation.z + localPosition.z);
113 
114  // std::cout << shapeVertex.position << std::endl;
115  shapeVertex.normal = vec3(currentCubeVertex.normal);
116 
117  shapeVertex.texCoords =
118  vec2(texture + currentCubeVertex.texCoords.x /
119  TextureOffsetEnum::NUMBER_TEXTURE,
120  currentCubeVertex.texCoords.y); // + offset
121  vertexList.push_back(shapeVertex);
122  }
